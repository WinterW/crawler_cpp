CREATE TABLE SEEDS (
  seed_url VARCHAR(200) NOT NULL,
  is_valid INT UNSIGNED NOT NULL DEFAULT 1,
  is_next_page INT UNSIGNED NOT NULL DEFAULT 0,
  extract_method INT UNSIGNED NOT NULL DEFAULT 0,
  url_template VARCHAR(100) NOT NULL,
  max_depth INT UNSIGNED NOT NULL DEFAULT 1,
  freq INT UNSIGNED NOT NULL DEFAULT 3600,
  max_freq INT UNSIGNED NOT NULL DEFAULT 3600,
  min_freq INT UNSIGNED NOT NULL DEFAULT 60,
  update_time TIMESTAMP NOT NULL,
  
  INDEX(seed_url)
);
